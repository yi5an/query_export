```python
import clickhouse_connect
from typing import AsyncIterator, Any, Dict, List
from backend.app.services.connector.base import BaseConnector


class ClickHouseConnector(BaseConnector):
    """ClickHouse 连接器"""

    def __init__(self, config: Dict[str, Any]):
        super().__init__(config)
        self._client = None

    async def connect(self) -> None:
        """建立连接"""
        self._client = clickhouse_connect.get_client(
            host=self.config.get('host'),
            port=self.config.get('port', 8123),
            database=self.config.get('database', 'default'),
            username=self.config.get('username', 'default'),
            password=self.config.get('password'),
            secure=self.config.get('extra_config', {}).get('ssl', False)
        )

    async def close(self) -> None:
        """关闭连接"""
        if self._client:
            self._client.close()

    async def test_connection(self) -> bool:
        """测试连接"""
        try:
            result = self._client.query("SELECT 1")
            return True
        except Exception:
            return False

    async def execute(self, sql: str, limit: int = None) -> Dict[str, Any]:
        """执行查询"""
        if limit:
            sql = f"{sql} LIMIT {limit}"

        result = self._client.query(sql)
        columns = result.column_names
        rows = [list(row) for row in result.result_rows]

        return {
            "columns": columns,
            "rows": rows,
            "row_count": len(rows)
        }

    async def stream_execute(self, sql: str, batch_size: int = 1000) -> AsyncIterator[List[Any]]:
        """流式执行"""
        result = self._client.query(sql, stream=True)
        batch = []
        for row in result.stream_results():
            batch.append(list(row))
            if len(batch) >= batch_size:
                yield batch
                batch = []
        if batch:
            yield batch

    def get_export_formats(self) -> List[str]:
        """支持的导出格式"""
        return ["csv", "excel", "sql"]
```

```python
import redis.asyncio as redis
from typing import AsyncIterator, Any, Dict, List
from backend.app.services.connector.base import BaseConnector


class RedisConnector(BaseConnector):
    """Redis 连接器"""

    def __init__(self, config: Dict[str, Any]):
        super().__init__(config)
        self._client = None

    async def connect(self) -> None:
        """建立连接"""
        extra_config = self.config.get('extra_config', {})

        # 检查是否为集群模式
        if 'cluster_nodes' in extra_config:
            # Redis Cluster
            self._client = redis.RedisCluster(
                nodes=extra_config['cluster_nodes'],
                password=self.config.get('password'),
                ssl=extra_config.get('ssl', False)
            )
        else:
            # 单机 Redis
            self._client = redis.Redis(
                host=self.config.get('host'),
                port=self.config.get('port', 6379),
                db=self.config.get('database', 0),
                password=self.config.get('password'),
                ssl=extra_config.get('ssl', False)
            )
        await self._client.ping()

    async def close(self) -> None:
        """关闭连接"""
        if self._client:
            await self._client.close()

    async def test_connection(self) -> bool:
        """测试连接"""
        try:
            await self._client.ping()
            return True
        except Exception:
            return False

    async def execute(self, command: str, limit: int = None) -> Dict[str, Any]:
        """
        执行 Redis 命令
        command: 如 "KEYS user:*" 或 "GET mykey"
        """
        parts = command.split()
        cmd = parts[0].upper()
        args = parts[1:]

        # 执行命令
        result = await self._client.execute_command(*[cmd] + args)

        # 转换为表格格式
        if isinstance(result, list):
            # 列表结果（如 KEYS, LRANGE 等）
            rows = [[item] for item in result[:limit] if limit else result]
            columns = ["result"]
        elif isinstance(result, dict):
            # 字典结果（如 HGETALL）
            rows = [[k, v] for k, v in result.items()]
            columns = ["key", "value"]
        else:
            # 单值结果
            rows = [[result]]
            columns = ["result"]

        return {
            "columns": columns,
            "rows": rows,
            "row_count": len(rows)
        }

    async def stream_execute(self, command: str, batch_size: int = 1000) -> AsyncIterator[List[Any]]:
        """流式执行（Redis 通常不需要流式）"""
        result = await self.execute(command)
        yield result["rows"]

    def get_export_formats(self) -> List[str]:
        """支持的导出格式"""
        return ["csv", "excel"]
```

```python
# backend/app/services/connector/elasticsearch.py
from elasticsearch import AsyncElasticsearch
from typing import AsyncIterator, Any, Dict, List
from backend.app.services.connector.base import BaseConnector


class ElasticsearchConnector(BaseConnector):
    """Elasticsearch 连接器"""

    def __init__(self, config: Dict[str, Any]):
        super().__init__(config)
        self._client = None

    async def connect(self) -> None:
        """建立连接"""
        extra_config = self.config.get('extra_config', {})

        self._client = AsyncElasticsearch(
            hosts=[f"{self.config['host']}:{self.config.get('port', 9200)}"],
            basic_auth=(
                (self.config.get('username'), self.config.get('password'))
            ) if self.config.get('username') else None,
            api_key=extra_config.get('api_key'),
            verify_ssl=extra_config.get('ssl', True),
        )
        await self._client.ping()

    async def close(self) -> None:
        """关闭连接"""
        if self._client:
            await self._client.close()

    async def test_connection(self) -> bool:
        """测试连接"""
        try:
            info = await self._client.info()
            return True
        except Exception:
            return False

    async def execute(self, query: str, limit: int = None) -> Dict[str, Any]:
        """
        执行查询
        query: JSON 格式的 Query DSL 或简单查询
        """
        import json

        extra_config = self.config.get('extra_config', {})
        index = extra_config.get('index_pattern', '*') or self.config.get('database', '*')

        # 尝试解析为 JSON DSL
        try:
            query_dsl = json.loads(query)
        except:
            # 简单查询字符串
            query_dsl = {"query": {"query_string": {"query": query}}}

        if limit:
            query_dsl.setdefault('size', limit)
        else:
            query_dsl.setdefault('size', 10)

        result = await self._client.search(index=index, body=query_dsl)

        # 提取结果
        columns = []
        rows = []

        for hit in result['hits']['hits']:
            source = hit['_source']
            if not columns:
                columns = list(source.keys())
            rows.append([source.get(col) for col in columns])

        return {
            "columns": columns,
            "rows": rows,
            "row_count": len(rows),
            "total": result['hits']['total']['value']
        }

    async def stream_execute(self, query: str, batch_size: int = 1000) -> AsyncIterator[List[Any]]:
        """流式执行"""
        # 使用 scroll API 或 pit for point in time
        extra_config = self.config.get('extra_config', {})
        index = extra_config.get('index_pattern', '*') or self.config.get('database', '*')

        import json
        try:
            query_dsl = json.loads(query)
        except:
            query_dsl = {"query": {"query_string": {"query": query}}}

        query_dsl['size'] = batch_size

        # 使用 scroll
        resp = await self._client.search(index=index, body=query_dsl, scroll='2m')
        scroll_id = resp['_scroll_id']

        while len(resp['hits']['hits']) > 0:
            batch = []
            columns = []

            for hit in resp['hits']['hits']:
                source = hit['_source']
                if not columns:
                    columns = list(source.keys())
                batch.append([source.get(col) for col in columns])

            yield batch

            if len(batch) < batch_size:
                break

            resp = await self._client.scroll(scroll_id=scroll_id, scroll='2m')

        await self._client.clear_scroll(scroll_id=scroll_id)

    def get_export_formats(self) -> List[str]:
        """支持的导出格式"""
        return ["csv", "excel", "json"]
```

```python
import asyncio
import mysql.connector
from typing import AsyncIterator, Any, Dict, List
from backend.app.services.connector.base import BaseConnector


class DorisConnector(BaseConnector):
    """Doris 连接器 (使用 MySQL 协议)"""

    def __init__(self, config: Dict[str, Any]):
        super().__init__(config)
        self._pool = None

    async def connect(self) -> None:
        """建立连接池"""
        # 注意: mysql-connector-python 是同步的，需要在 asyncio 中运行
        from databases import Database

        database_url = (
            f"mysql://{self.config.get('username', 'root')}:"
            f"{self.config.get('password', '')}@"
            f"{self.config.get('host')}:"
            f"{self.config.get('port', 9030)}/"
            f"{self.config.get('database', '')}"
        )

        self._pool = Database(database_url)
        await self._pool.connect()

    async def close(self) -> None:
        """关闭连接"""
        if self._pool:
            await self._pool.disconnect()

    async def test_connection(self) -> bool:
        """测试连接"""
        try:
            await self._pool.execute("SELECT 1")
            return True
        except Exception:
            return False

    async def execute(self, sql: str, limit: int = None) -> Dict[str, Any]:
        """执行查询"""
        if limit:
            sql = f"{sql} LIMIT {limit}"

        rows = await self._pool.fetch_all(sql)
        if rows:
            columns = list(rows[0].keys())
            row_data = [list(row.values()) for row in rows]
        else:
            columns = []
            row_data = []

        return {
            "columns": columns,
            "rows": row_data,
            "row_count": len(row_data)
        }

    async def stream_execute(self, sql: str, batch_size: int = 1000) -> AsyncIterator[List[Any]]:
        """流式执行"""
        # 使用 fetchmany 实现流式
        async with self._pool.connection() as conn:
            async for batch in conn.iterate(sql, batch_size):
                yield [list(row.values()) for row in batch]

    def get_export_formats(self) -> List[str]:
        """支持的导出格式"""
        return ["csv", "excel", "sql"]
```

```python
from minio import Minio
from typing import AsyncIterator, Any, Dict, List
from backend.app.services.connector.base import BaseConnector


class MinioConnector(BaseConnector):
    """MinIO 对象存储连接器"""

    def __init__(self, config: Dict[str, Any]):
        super().__init__(config)
        self._client = None

    async def connect(self) -> None:
        """建立连接"""
        extra_config = self.config.get('extra_config', {})

        self._client = Minio(
            f"{self.config['host']}:{self.config.get('port', 9000)}",
            access_key=self.config.get('username'),
            secret_key=self.config.get('password'),
            secure=extra_config.get('secure', False)
        )
        # 测试连接
        self._client.list_buckets()

    async def close(self) -> None:
        """MinIO 客户端不需要显式关闭"""

    async def test_connection(self) -> bool:
        """测试连接"""
        try:
            self._client.list_buckets()
            return True
        except Exception:
            return False

    async def execute(self, path: str, limit: int = None) -> Dict[str, Any]:
        """
        列出对象
        path: bucket/prefix 格式
        """
        parts = path.split('/', 1)
        bucket = parts[0]
        prefix = parts[1] if len(parts) > 1 else ''

        objects = self._client.list_objects(bucket, prefix=prefix, recursive=False)

        columns = ["name", "size", "last_modified", "etag"]
        rows = []

        for obj in objects:
            if limit and len(rows) >= limit:
                break
            rows.append([
                obj.object_name,
                obj.size,
                obj.last_modified,
                obj.etag
            ])

        return {
            "columns": columns,
            "rows": rows,
            "row_count": len(rows)
        }

    async def stream_execute(self, path: str, batch_size: int = 1000) -> AsyncIterator[List[Any]]:
        """流式列出对象"""
        parts = path.split('/', 1)
        bucket = parts[0]
        prefix = parts[1] if len(parts) > 1 else ''

        batch = []
        for obj in self._client.list_objects(bucket, prefix=prefix, recursive=True):
            batch.append([obj.object_name, obj.size, obj.last_modified, obj.etag])
            if len(batch) >= batch_size:
                yield batch
                batch = []
        if batch:
            yield batch

    def get_export_formats(self) -> List[str]:
        """MinIO 不支持传统导出，直接下载文件"""
        return []

    def get_presigned_url(self, bucket: str, object_name: str, expires: int = 3600) -> str:
        """获取预签名下载链接"""
        return self._client.presigned_get_object(bucket, object_name, expires=expires)
```

```python
from typing import Dict, Any
from backend.app.services.connector.base import BaseConnector
from backend.app.services.connector.postgres import PostgresConnector
from backend.app.services.connector.clickhouse import ClickHouseConnector
from backend.app.services.connector.redis import RedisConnector
from backend.app.services.connector.elasticsearch import ElasticsearchConnector
from backend.app.services.connector.doris import DorisConnector
from backend.app.services.connector.minio import MinioConnector

CONNECTOR_REGISTRY = {
    'postgres': PostgresConnector,
    'clickhouse': ClickHouseConnector,
    'doris': DorisConnector,
    'redis': RedisConnector,
    'redis-cluster': RedisConnector,  # 使用相同的 Redis 连接器
    'elasticsearch': ElasticsearchConnector,
    'minio': MinioConnector,
}


def get_connector(datasource_type: str, config: Dict[str, Any]) -> BaseConnector:
    """根据数据源类型获取连接器实例"""
    connector_class = CONNECTOR_REGISTRY.get(datasource_type)
    if not connector_class:
        supported = ', '.join(CONNECTOR_REGISTRY.keys())
        raise ValueError(
            f"Unsupported datasource type: {datasource_type}. "
            f"Supported types: {supported}"
        )
    return connector_class(config)


def get_supported_types() -> List[str]:
    """获取支持的数据源类型列表"""
    return list(CONNECTOR_REGISTRY.keys())
```

```python
# backend/app/api/v1/saved_sqls.py
from fastapi import APIRouter, Depends, HTTPException, Query
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, or_
from typing import List, Optional
from backend.app.api.deps import get_db
from backend.app.schemas.saved_sql import SavedSqlCreate, SavedSqlUpdate, SavedSql as SavedSqlSchema
from backend.app.models.saved_sql import SavedSql as SavedSqlModel
from datetime import datetime

router = APIRouter(prefix="/saved-sqls", tags=["saved-sqls"])


@router.get("/", response_model=List[SavedSqlSchema])
async def list_saved_sqls(
    datasource_id: Optional[int] = None,
    search: Optional[str] = None,
    tags: Optional[str] = None,  # 逗号分隔的标签
    db: AsyncSession = Depends(get_db)
):
    """获取保存的 SQL 列表，支持搜索和标签筛选"""
    query = select(SavedSqlModel)

    if datasource_id:
        query = query.where(SavedSqlModel.datasource_id == datasource_id)

    if search:
        # 在名称、注释、SQL文本中搜索
        query = query.where(
            or_(
                SavedSqlModel.name.ilike(f"%{search}%"),
                SavedSqlModel.comment.ilike(f"%{search}%"),
                SavedSqlModel.sql_text.ilike(f"%{search}%")
            )
        )

    if tags:
        tag_list = tags.split(',')
        # 使用 JSON 数组包含查询
        for tag in tag_list:
            query = query.where(SavedSqlModel.tags.contains([tag.strip()]))

    query = query.order_by(SavedSqlModel.updated_at.desc())

    result = await db.execute(query)
    return result.scalars().all()


@router.get("/{sql_id}", response_model=SavedSqlSchema)
async def get_saved_sql(sql_id: int, db: AsyncSession = Depends(get_db)):
    """获取单个保存的 SQL"""
    result = await db.execute(
        select(SavedSqlModel).where(SavedSqlModel.id == sql_id)
    )
    sql = result.scalar_one_or_none()
    if not sql:
        raise HTTPException(status_code=404, detail="Saved SQL not found")
    return sql


@router.post("/", response_model=SavedSqlSchema)
async def create_saved_sql(
    sql: SavedSqlCreate,
    db: AsyncSession = Depends(get_db)
):
    """保存 SQL"""
    # 生成 embedding（Phase 8 实现）
    db_sql = SavedSqlModel(**sql.dict())
    db.add(db_sql)
    await db.commit()
    await db.refresh(db_sql)
    return db_sql


@router.put("/{sql_id}", response_model=SavedSqlSchema)
async def update_saved_sql(
    sql_id: int,
    sql: SavedSqlUpdate,
    db: AsyncSession = Depends(get_db)
):
    """更新保存的 SQL"""
    result = await db.execute(
        select(SavedSqlModel).where(SavedSqlModel.id == sql_id)
    )
    db_sql = result.scalar_one_or_none()
    if not db_sql:
        raise HTTPException(status_code=404, detail="Saved SQL not found")

    update_data = sql.dict(exclude_unset=True)
    for field, value in update_data.items():
        setattr(db_sql, field, value)

    await db.commit()
    await db.refresh(db_sql)
    return db_sql


@router.delete("/{sql_id}")
async def delete_saved_sql(sql_id: int, db: AsyncSession = Depends(get_db)):
    """删除保存的 SQL"""
    result = await db.execute(
        select(SavedSqlModel).where(SavedSqlModel.id == sql_id)
    )
    db_sql = result.scalar_one_or_none()
    if not db_sql:
        raise HTTPException(status_code=404, detail="Saved SQL not found")

    await db.delete(db_sql)
    await db.commit()
    return {"message": "Saved SQL deleted"}


@router.post("/{sql_id}/run", response_model=dict)
async def run_saved_sql(
    sql_id: int,
    limit: int = 10,
    db: AsyncSession = Depends(get_db)
):
    """运行保存的 SQL"""
    # 获取保存的 SQL
    result = await db.execute(
        select(SavedSqlModel).where(SavedSqlModel.id == sql_id)
    )
    saved_sql = result.scalar_one_or_none()
    if not saved_sql:
        raise HTTPException(status_code=404, detail="Saved SQL not found")

    # 更新运行次数
    saved_sql.run_count += 1
    saved_sql.last_run_at = datetime.now()
    await db.commit()

    # 执行查询
    from backend.app.services.query import execute_query
    from backend.app.models.datasource import Datasource as DatasourceModel

    # 获取数据源
    ds_result = await db.execute(
        select(DatasourceModel).where(DatasourceModel.id == saved_sql.datasource_id)
    )
    datasource = ds_result.scalar_one_or_none()

    config = {
        "host": datasource.host,
        "port": datasource.port,
        "database": datasource.database,
        "username": datasource.username,
        "password_encrypted": datasource.password_encrypted,
    }
    if datasource.extra_config:
        config.update(datasource.extra_config)

    query_result = await execute_query(
        datasource.type,
        config,
        saved_sql.sql_text,
        limit
    )

    return {
        "saved_sql_id": sql_id,
        "run_count": saved_sql.run_count,
```

```python
from abc import ABC, abstractmethod
from typing import List


class BaseAIProvider(ABC):
    """AI Provider 抽象基类"""

    def __init__(self, api_key: str, base_url: str = None, model: str = None):
        self.api_key = api_key
        self.base_url = base_url
        self.model = model

    @abstractmethod
    async def generate_embedding(self, text: str) -> List[float]:
        """生成文本向量，用于语义相似匹配"""
        pass

    @abstractmethod
    async def generate_sql(
        self,
        description: str,
        context: dict = None
    ) -> str:
        """
        根据描述生成 SQL
        context: 可包含表结构、相似 SQL 示例等
        """
        pass
```

```python
from backend.app.models.ai_config import AiConfig as AiConfigModel
from backend.app.models.saved_sql import SavedSql as SavedSqlModel

async def get_active_ai_provider(db: AsyncSession):
    """获取激活的 AI Provider"""
    result = await db.execute(
        select(AiConfigModel).where(AiConfigModel.is_active == True)
    )
    config = result.scalar_one_or_none()
    if not config:
        raise ValueError("No active AI configuration found")

    return get_ai_provider(
        config.provider,
        decrypt_password(config.api_key_encrypted),
        config.base_url,
        config.model_name
    )


async def find_similar_sqls(
    description: str,
    datasource_id: int,
    embedding: List[float],
    threshold: float = 0.3,
    limit: int = 3,
    db: AsyncSession = None
) -> List[SavedSqlModel]:
    """
    使用向量相似搜索找到相似的 SQL
    threshold: 余弦距离阈值，越小越相似
    """
    if not db:
        return []

    # 计算余弦距离
    distance = CosineDistance('embedding', embedding)

    result = await db.execute(
        select(SavedSqlModel)
        .where(SavedSqlModel.datasource_id == datasource_id)
        .where(SavedSqlModel.embedding.isnot(None))
        .order_by(distance)
        .limit(limit)
    )
    return result.scalars().all()


async def generate_sql_with_ai(
    description: str,
    datasource_id: int,
    db: AsyncSession
) -> dict:
    """使用 AI 生成 SQL"""
    # 获取 AI Provider
    provider = await get_active_ai_provider(db)

    # 生成描述的 embedding
    description_embedding = await provider.generate_embedding(description)

    # 查找相似的 SQL
    similar_sqls = await find_similar_sqls(
        description,
        datasource_id,
        description_embedding,
        db=db
    )

    # 构建上下文
    context = {}
    if similar_sqls:
        similar_sql = similar_sqls[0]
        context['similar_sql'] = f"-- {similar_sql.comment}\n{similar_sql.sql_text}"
        # 可以添加表结构等信息
        # context['table_structure'] = ...

    # 生成 SQL
    generated_sql = await provider.generate_sql(description, context)

    return {
        "sql": generated_sql,
        "matched_sql": {
            "id": similar_sqls[0].id if similar_sqls else None,
            "name": similar_sqls[0].name if similar_sqls else None,
            "comment": similar_sqls[0].comment if similar_sqls else None,
        } if similar_sqls else None
    }

        tags=tags,
        embedding=embedding
    )
```

### Task 25: 真正的流式导出

**说明**: 当前导出服务将所有行加载到内存，对于大表会有问题。实现真正的流式处理。

**文件:**

```python
# backend/app/services/export/streaming.py
import asyncio
import uuid
from pathlib import Path
from typing import AsyncIterator
from backend.app.services.connector.base import BaseConnector
from backend.app.services.export.csv_handler import CSVHandler
from backend.app.services.export.excel_handler import ExcelHandler

EXPORT_HANDLERS = {
    "csv": CSVHandler,
    "excel": ExcelHandler,
}


async def stream_export(
    connector: BaseConnector,
    sql: str,
    export_format: str,
    output_path: str,
    batch_size: int = 1000,
    progress_callback=None
) -> int:
    """
    真正的流式导出
    直接从连接器流式读取并写入文件，不积累在内存中
    """
    handler_class = EXPORT_HANDLERS.get(export_format)
    if not handler_class:
        raise ValueError(f"Unsupported format: {export_format}")

    handler = handler_class()
    output_file = Path(output_path)
    output_file.parent.mkdir(parents=True, exist_ok=True)

    total_rows = 0

    # 获取第一批数据以获取列名
    first_batch = await anext(connector.stream_execute(sql, batch_size), None)
    if not first_batch:
        return 0

    # 打开文件准备写入
    if export_format == 'csv':
        import csv
        with open(output_file, 'w', newline='', encoding='utf-8') as f:
            writer = csv.writer(f)

            # 写入第一批
            columns = [f"col{i}" for i in range(len(first_batch[0]))]
            writer.writerow(columns)
            writer.writerows(first_batch)
            total_rows += len(first_batch)

            # 继续流式处理
            async for batch in connector.stream_execute(sql, batch_size):
                writer.writerows(batch)
                total_rows += len(batch)

                if progress_callback:
                    await progress_callback(total_rows)

    elif export_format == 'excel':
        import xlsxwriter
        workbook = xlsxwriter.Workbook(output_file, {'in_memory': False})
        worksheet = workbook.add_worksheet()

        row_idx = 0

        # 写入第一批
        for col_idx, value in enumerate(first_batch[0]):
            worksheet.write(row_idx, col_idx, f"col{col_idx}")
        row_idx += 1

        for row_data in first_batch:
            for col_idx, value in enumerate(row_data):
                worksheet.write(row_idx, col_idx, value)
            row_idx += 1
        total_rows += len(first_batch)

        # 继续流式处理
        async for batch in connector.stream_execute(sql, batch_size):
            for row_data in batch:
                for col_idx, value in enumerate(row_data):
                    worksheet.write(row_idx, col_idx, value)
                row_idx += 1
            total_rows += len(batch)

            if progress_callback:
                await progress_callback(total_rows)

        workbook.close()

    return total_rows
```
